    <!-- JavaScript for Countdown Timer and Servers -->
    <script>
        const countdown = () => {
            // Set the date for the annual competition (July 1st)
            const now = new Date();
            let targetYear = now.getFullYear();
            const competitionDate = new Date(targetYear, 6, 1, 0, 0, 0); // Month is 0-indexed (July is 6)

            // If the current date is after July 1st, set the target for next year
            if (now.getTime() > competitionDate.getTime()) {
                targetYear++;
                competitionDate.setFullYear(targetYear);
            }

            const daysEl = document.getElementById('days');
            const hoursEl = document.getElementById('hours');
            const minutesEl = document.getElementById('minutes');
            const secondsEl = document.getElementById('seconds');
            const countdownMessageEl = document.getElementById('countdown-message');
            const countdownTimerEl = document.getElementById('countdown-timer');

            const updateCountdown = () => {
                const now = new Date().getTime();
                const timeLeft = competitionDate.getTime() - now;

                if (timeLeft < 0) {
                    countdownTimerEl.classList.add('hidden');
                    countdownMessageEl.classList.remove('hidden');
                    clearInterval(interval);
                    return;
                }

                // Calculate days, hours, minutes, and seconds
                const days = Math.floor(timeLeft / (1000 * 60 * 60 * 24));
                const hours = Math.floor((timeLeft % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((timeLeft % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((timeLeft % (1000 * 60)) / 1000);

                // Display the results
                daysEl.innerText = String(days).padStart(2, '0');
                hoursEl.innerText = String(hours).padStart(2, '0');
                minutesEl.innerText = String(minutes).padStart(2, '0');
                secondsEl.innerText = String(seconds).padStart(2, '0');
            };

            // Update the countdown every second
            const interval = setInterval(updateCountdown, 1000);
            updateCountdown(); // Initial call to display the countdown immediately
        };
        
        window.onload = () => {
            countdown();
        };
    </script>
